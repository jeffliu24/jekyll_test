#include <stdio.h>

int sort(int a[10])
{
    int i,j;

    for (i=0; i<10; i++)
    {
        for (j=9; j>i; j--)
        {
            if (a[i] < a[j])
            {
                a[i] = a[i] + a[j];
                a[j] = a[i] - a[j];
                a[i] = a[i] - a[j];
            }
        }
    }

    return 0;
}

int sort_n(int a[], int n)
{
    int i,j;

    for (i=0; i<n; i++)
    {
        for (j=n-1; j>i; j--)
        {
            if (a[i] < a[j])
            {
                a[i] = a[i] + a[j];
                a[j] = a[i] - a[j];
                a[i] = a[i] - a[j];
            }
        }
    }

    return 0;
}
